  "customerId": 1 // 客户ID

  "deliverAddress": "xxx" // 邮寄地址

  "contactName": "xxx" // 联系人
  "mobile": "xxx" // 手机

    "name": "xxx" // 发票名头
    "taxCode": "xxx" // 发票税号
    "address": "xxx" // 地址
    "mobile": "xxx" // 联系电话
    "bankName": "xxx" // 开户行

    "name": "xxx" // 发票名头
